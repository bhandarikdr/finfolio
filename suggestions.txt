@Database(
    entities = [IpoMaster::class, BoidEntry::class, DpMaster::class, ResultCache::class],
    version = 1,
    exportSchema = false
)
abstract class FinFolioDatabase : RoomDatabase() {
    abstract fun ipoDao(): IpoDao
    abstract fun boidDao(): BoidDao
    abstract fun dpDao(): DpDao
    abstract fun resultCacheDao(): ResultCacheDao

    companion object {
        @Volatile
        private var INSTANCE: FinFolioDatabase? = null

        fun getInstance(context: Context): FinFolioDatabase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    FinFolioDatabase::class.java,
                    "finfolio_db"
                ).build()
                INSTANCE = instance
                instance
            }
        }
    }
}

// repository/MeroShareRepository.kt
class MeroShareRepository(
    private val context: Context,
    private val boidDao: BoidDao,
    private val dpDao: DpDao,
    private val resultCacheDao: ResultCacheDao
) {
    private val client = OkHttpClient.Builder()
        .connectTimeout(30, TimeUnit.SECONDS)
        .readTimeout(30, TimeUnit.SECONDS)
        .writeTimeout(30, TimeUnit.SECONDS)
        .addInterceptor { chain ->
            val request = chain.request().newBuilder()
                .addHeader("User-Agent", "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36")
                .addHeader("Accept", "application/json")
                .addHeader("Content-Type", "application/json")
                .build()
            chain.proceed(request)
        }
        .build()

    private val gson = Gson()
    
    companion object {
        private const val BASE_URL = "https://api.meroshare.cdsc.com.np"
        private const val DP_MASTER_URL = "https://www.sharesansar.com/dp-member-list"
        private const val RATE_LIMIT_DELAY = 1500L // 1.5 seconds between requests
    }

    // DP Mapping Functions
    suspend fun fetchDpMaster(): Result<List<DpMaster>> {
        return withContext(Dispatchers.IO) {
            try {
                val response = client.newCall(
                    Request.Builder().url(DP_MASTER_URL).build()
                ).execute()
                
                if (!response.isSuccessful) {
                    return@withContext Result.failure(Exception("Failed to fetch DP list"))
                }
                
                val html = response.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                val dpList = parseDpFromHtml(html)
                
                dpList.forEach { dp ->
                    dpDao.insertDp(dp)
                }
                
                Result.success(dpList)
            } catch (e: Exception) {
                // Fallback to hardcoded mapping
                val fallbackDps = getFallbackDpMapping()
                fallbackDps.forEach { dp ->
                    dpDao.insertDp(dp)
                }
                Result.success(fallbackDps)
            }
        }
    }
    
    private fun parseDpFromHtml(html: String): List<DpMaster> {
        val dps = mutableListOf<DpMaster>()
        // Parse HTML table - simplified regex example
        val pattern = Regex("""<tr>.*?<td>(\d+)</td>.*?<td>(.*?)</td>.*?</tr>""", RegexOption.DOT_MATCHES_ALL)
        pattern.findAll(html).forEach { match ->
            val code = match.groupValues[1]
            val name = match.groupValues[2]
            // Generate clientId from DP code (actual mapping logic needed)
            val clientId = getClientIdFromDpCode(code)
            dps.add(DpMaster(dpCode = code, clientId = clientId, name = name))
        }
        return dps
    }
    
    private fun getFallbackDpMapping(): List<DpMaster> {
        // Hardcoded fallback for common DP codes
        return listOf(
            DpMaster("10600", "173", "NIBL Ace Capital"),
            DpMaster("10400", "47", "NMB Capital"),
            DpMaster("10200", "3", "Laxmi Capital")
            // Add more as needed
        )
    }
    
    private fun getClientIdFromDpCode(dpCode: String): String {
        // You'll need to implement the actual mapping logic
        // This is a simplified version
        return when (dpCode) {
            "10600" -> "173"
            "10400" -> "47"
            "10200" -> "3"
            else -> "173" // Default fallback
        }
    }

    // Authentication
    private suspend fun authenticate(username: String, password: String): Result<String> {
        return withContext(Dispatchers.IO) {
            try {
                val json = mapOf(
                    "username" to username,
                    "password" to password
                )
                
                val body = RequestBody.create(
                    MediaType.parse("application/json")!!,
                    gson.toJson(json)
                )
                
                val request = Request.Builder()
                    .url("$BASE_URL/auth/")
                    .post(body)
                    .build()
                
                val response = client.newCall(request).execute()
                
                if (response.isSuccessful) {
                    val responseBody = response.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                    val responseMap = gson.fromJson(responseBody, Map::class.java)
                    val token = responseMap["token"] as? String
                    
                    if (token != null) {
                        Result.success(token)
                    } else {
                        Result.failure(Exception("Invalid credentials"))
                    }
                } else {
                    Result.failure(Exception("Authentication failed: ${response.code}"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    // Get bank details
    private suspend fun getBankDetails(token: String): Result<Map<String, Any>> {
        return withContext(Dispatchers.IO) {
            try {
                val request = Request.Builder()
                    .url("$BASE_URL/bank/")
                    .addHeader("Authorization", "Bearer $token")
                    .get()
                    .build()
                
                val response = client.newCall(request).execute()
                
                if (response.isSuccessful) {
                    val responseBody = response.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                    val responseMap = gson.fromJson(responseBody, Map::class.java)
                    
                    // Parse bank details
                    val details = mutableMapOf<String, Any>()
                    if (responseMap["data"] is List<*>) {
                        val bankData = (responseMap["data"] as List<*>).firstOrNull()
                        if (bankData is Map<*, *>) {
                            details["accountBranchId"] = (bankData["accountBranchId"] as? Double)?.toInt() ?: 0
                            details["bankId"] = (bankData["bankId"] as? Double)?.toInt() ?: 0
                        }
                    }
                    Result.success(details)
                } else {
                    Result.failure(Exception("Failed to get bank details"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    // Check allotment status
    suspend fun checkAllotment(
        boid: String,
        username: String,
        password: String,
        companyName: String,
        resultPortalId: String
    ): Result<ResultCache> {
        return withContext(Dispatchers.IO) {
            try {
                // Authenticate first
                val authResult = authenticate(username, password)
                if (authResult.isFailure) {
                    return@withContext Result.failure(authResult.exceptionOrNull() ?: Exception("Authentication failed"))
                }
                val token = authResult.getOrNull() ?: return@withContext Result.failure(Exception("No token received"))
                
                // Get bank details
                val bankResult = getBankDetails(token)
                if (bankResult.isFailure) {
                    return@withContext Result.failure(bankResult.exceptionOrNull() ?: Exception("Failed to get bank details"))
                }
                val bankDetails = bankResult.getOrNull() ?: return@withContext Result.failure(Exception("No bank details"))
                
                // Submit search
                val searchJson = mapOf(
                    "companyShareId" to resultPortalId,
                    "boid" to boid,
                    "accountBranchId" to (bankDetails["accountBranchId"] ?: 0)
                )
                
                val searchBody = RequestBody.create(
                    MediaType.parse("application/json")!!,
                    gson.toJson(searchJson)
                )
                
                val searchRequest = Request.Builder()
                    .url("$BASE_URL/active/search/")
                    .addHeader("Authorization", "Bearer $token")
                    .post(searchBody)
                    .build()
                
                val searchResponse = client.newCall(searchRequest).execute()
                
                if (searchResponse.isSuccessful) {
                    val responseBody = searchResponse.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                    val responseMap = gson.fromJson(responseBody, Map::class.java)
                    
                    // Parse result
                    val status = when {
                        responseMap.toString().contains("allotted", ignoreCase = true) -> "Allotted"
                        responseMap.toString().contains("not allotted", ignoreCase = true) -> "Not Allotted"
                        else -> "Pending"
                    }
                    
                    // Extract units if allotted
                    val units = if (status == "Allotted") {
                        val unitsPattern = Regex("""(\d+)\s*units?""", RegexOption.IGNORE_CASE)
                        unitsPattern.find(responseMap.toString())?.groupValues?.get(1)?.toIntOrNull() ?: 0
                    } else 0
                    
                    val result = ResultCache(
                        companyName = companyName,
                        boid = boid,
                        status = status,
                        units = units,
                        checkedAt = System.currentTimeMillis()
                    )
                    
                    // Save to cache
                    resultCacheDao.insertResult(result)
                    
                    Result.success(result)
                } else {
                    Result.failure(Exception("Search failed: ${searchResponse.code}"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    // Apply for IPO
    suspend fun applyForIpo(
        boid: String,
        username: String,
        password: String,
        crn: String,
        pin: String,
        ipoName: String,
        issueId: String,
        units: Int = 10
    ): Result<String> {
        return withContext(Dispatchers.IO) {
            try {
                // Authenticate
                val authResult = authenticate(username, password)
                if (authResult.isFailure) {
                    return@withContext Result.failure(authResult.exceptionOrNull() ?: Exception("Authentication failed"))
                }
                val token = authResult.getOrNull() ?: return@withContext Result.failure(Exception("No token received"))
                
                // Get bank details
                val bankResult = getBankDetails(token)
                if (bankResult.isFailure) {
                    return@withContext Result.failure(bankResult.exceptionOrNull() ?: Exception("Failed to get bank details"))
                }
                val bankDetails = bankResult.getOrNull() ?: return@withContext Result.failure(Exception("No bank details"))
                
                // Get applicable issues
                val issuesResult = getApplicableIssues(token)
                if (issuesResult.isFailure) {
                    return@withContext Result.failure(issuesResult.exceptionOrNull() ?: Exception("Failed to get issues"))
                }
                val issues = issuesResult.getOrNull() ?: return@withContext Result.failure(Exception("No issues found"))
                
                // Find the specific issue
                val issue = issues.find { it["companyName"] == ipoName }
                    ?: return@withContext Result.failure(Exception("IPO not found or not applicable"))
                
                val shareId = (issue["id"] as? Double)?.toInt() ?: return@withContext Result.failure(Exception("Invalid share ID"))
                val minUnits = (issue["minUnits"] as? Double)?.toInt() ?: 10
                
                // Apply
                val applyJson = mapOf(
                    "shareId" to shareId,
                    "appliedUnits" to (if (units > minUnits) units else minUnits),
                    "accountBranchId" to (bankDetails["accountBranchId"] ?: 0),
                    "crn" to crn,
                    "pin" to pin
                )
                
                val applyBody = RequestBody.create(
                    MediaType.parse("application/json")!!,
                    gson.toJson(applyJson)
                )
                
                val applyRequest = Request.Builder()
                    .url("$BASE_URL/share/apply")
                    .addHeader("Authorization", "Bearer $token")
                    .post(applyBody)
                    .build()
                
                val applyResponse = client.newCall(applyRequest).execute()
                
                if (applyResponse.isSuccessful) {
                    val responseBody = applyResponse.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                    val responseMap = gson.fromJson(responseBody, Map::class.java)
                    val message = (responseMap["message"] as? String) ?: "Application submitted"
                    Result.success(message)
                } else {
                    Result.failure(Exception("Application failed: ${applyResponse.code}"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    private suspend fun getApplicableIssues(token: String): Result<List<Map<String, Any>>> {
        return withContext(Dispatchers.IO) {
            try {
                val request = Request.Builder()
                    .url("$BASE_URL/applicableIssue/")
                    .addHeader("Authorization", "Bearer $token")

                    .build()
                
                val response = client.newCall(request).execute()
                
                if (response.isSuccessful) {
                    val responseBody = response.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
                    val responseMap = gson.fromJson(responseBody, Map::class.java)
                    val data = responseMap["data"] as? List<Map<String, Any>> ?: emptyList()
                    Result.success(data)
                } else {
                    Result.failure(Exception("Failed to get issues"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }

// viewmodel/BulkIpoViewModel.kt
class BulkIpoViewModel(
    private val ipoDao: IpoDao,
    private val boidDao: BoidDao,
    private val dpDao: DpDao,
    private val resultCacheDao: ResultCacheDao,
    private val meroShareRepo: MeroShareRepository,
    private val coroutineScope: CoroutineScope
) : ViewModel() {
    
    private val _uiState = MutableStateFlow(BulkIpoUiState())
    val uiState: StateFlow<BulkIpoUiState> = _uiState.asStateFlow()
    
    private val _progressLog = MutableStateFlow<List<String>>(emptyList())
    val progressLog: StateFlow<List<String>> = _progressLog.asStateFlow()
    
    private var isProcessing = false
    
    init {
        loadInitialData()
    }
    
    private fun loadInitialData() {
        viewModelScope.launch {
            try {
                val ipos = ipoDao.getAllIpos()
                val boids = boidDao.getAllBoids()
                val dpCodes = dpDao.getAllDps()
                
                _uiState.value = _uiState.value.copy(
                    ipos = ipos,
                    boids = boids,
                    dpCodes = dpCodes,
                    isLoading = false
                )
            } catch (e: Exception) {
                _uiState.value = _uiState.value.copy(
                    error = e.message ?: "Failed to load data",
                    isLoading = false
                )
            }
        }
    }
    
    fun selectIpo(ipo: IpoMaster) {
        _uiState.value = _uiState.value.copy(selectedIpo = ipo)
        // Load cached results for this IPO
        viewModelScope.launch {
            val results = resultCacheDao.getResultsForIpo(ipo.companyName)
            _uiState.value = _uiState.value.copy(
                cachedResults = results.associateBy { it.boid }
            )
        }
    }
    
    fun toggleBoidSelection(boid: String) {
        val currentSelections = _uiState.value.selectedBoids.toMutableSet()
        if (currentSelections.contains(boid)) {
            currentSelections.remove(boid)
        } else {
            currentSelections.add(boid)
        }
        _uiState.value = _uiState.value.copy(selectedBoids = currentSelections)
    }
    
    fun selectAllBoids() {
        val boids = _uiState.value.boids.map { it.boid }.toSet()
        _uiState.value = _uiState.value.copy(selectedBoids = boids)
    }
    
    fun deselectAllBoids() {
        _uiState.value = _uiState.value.copy(selectedBoids = emptySet())
    }
    
    fun startBulkCheck() {
        if (isProcessing) return
        
        val selectedIpo = _uiState.value.selectedIpo
        if (selectedIpo == null) {
            addLog("Error: No IPO selected")
            return
        }
        
        val selectedBoids = _uiState.value.boids.filter { 
            it.boid in _uiState.value.selectedBoids && it.isEnabled 
        }
        
        if (selectedBoids.isEmpty()) {
            addLog("Error: No accounts selected or enabled")
            return
        }
        
        isProcessing = true
        _uiState.value = _uiState.value.copy(isProcessing = true)
        addLog("Starting bulk check for: ${selectedIpo.companyName}")
        addLog("Accounts to check: ${selectedBoids.size}")
        
        viewModelScope.launch {
            try {
                selectedBoids.forEachIndexed { index, boid ->
                    addLog("[${index + 1}/${selectedBoids.size}] Checking: ${boid.boid.takeLast(4)}")
                    
                    val result = meroShareRepo.checkAllotment(
                        boid = boid.boid,
                        username = boid.username ?: "",
                        password = boid.password ?: "",
                        companyName = selectedIpo.companyName,
                        resultPortalId = selectedIpo.resultPortalId ?: ""
                    )
                    
                    when (result) {
                        is Result.Success -> {
                            val cache = result.getOrNull()
                            if (cache != null) {
                                addLog("  ✓ Status: ${cache.status}${if (cache.status == "Allotted") " (${cache.units} units)" else ""}")
                                // Update UI
                                val updatedResults = _uiState.value.cachedResults.toMutableMap()
                                updatedResults[boid.boid] = cache
                                _uiState.value = _uiState.value.copy(
                                    cachedResults = updatedResults
                                )
                            }
                        }
                        is Result.Failure -> {
                            val error = result.exceptionOrNull()?.message ?: "Unknown error"
                            addLog("  ✗ Error: $error")
                        }
                    }
                    
                    // Rate limiting
                    delay(RATE_LIMIT_DELAY)
                }
                
                addLog("✅ Bulk check completed")
            } catch (e: Exception) {
                addLog("❌ Error: ${e.message}")
            } finally {
                isProcessing = false
                _uiState.value = _uiState.value.copy(isProcessing = false)
            }
        }
    }
    
    fun startBulkApply() {
        if (isProcessing) return
        
        val selectedIpo = _uiState.value.selectedIpo
        if (selectedIpo == null) {
            addLog("Error: No IPO selected")
            return
        }
        
        val selectedBoids = _uiState.value.boids.filter { 
            it.boid in _uiState.value.selectedBoids && it.isEnabled 
        }
        
        if (selectedBoids.isEmpty()) {
            addLog("Error: No accounts selected or enabled")
            return
        }
        
        // Confirm total cost
        val totalCost = selectedBoids.size * 1000 // Assuming 10 units * 100 per share
        addLog("💰 Total cost for ${selectedBoids.size} accounts: Rs. $totalCost")
        
        // Show confirmation dialog (handled in UI)
        _uiState.value = _uiState.value.copy(showApplyConfirmation = true)
    }
    
    fun confirmApply() {
        val selectedIpo = _uiState.value.selectedIpo
        if (selectedIpo == null) return
        
        val selectedBoids = _uiState.value.boids.filter { 
            it.boid in _uiState.value.selectedBoids && it.isEnabled 
        }
        
        isProcessing = true
        _uiState.value = _uiState.value.copy(
            isProcessing = true,
            showApplyConfirmation = false
        )
        
        addLog("Starting bulk apply for: ${selectedIpo.companyName}")
        addLog("Applying for ${selectedBoids.size} accounts")
        
        viewModelScope.launch {
            try {
                selectedBoids.forEachIndexed { index, boid ->
                    addLog("[${index + 1}/${selectedBoids.size}] Applying for: ${boid.boid.takeLast(4)}")
                    
                    val result = meroShareRepo.applyForIpo(
                        boid = boid.boid,
                        username = boid.username ?: "",
                        password = boid.password ?: "",
                        crn = boid.crn ?: "",
                        pin = boid.pin ?: "",
                        ipoName = selectedIpo.companyName,
                        issueId = selectedIpo.resultPortalId ?: ""
                    )
                    
                    when (result) {
                        is Result.Success -> {
                            addLog("  ✓ ${result.getOrNull()}")
                        }
                        is Result.Failure -> {
                            val error = result.exceptionOrNull()?.message ?: "Unknown error"
                            addLog("  ✗ Error: $error")
                        }
                    }
                    
                    delay(RATE_LIMIT_DELAY)
                }
                
                addLog("✅ Bulk apply completed")
            } catch (e: Exception) {
                addLog("❌ Error: ${e.message}")
            } finally {
                isProcessing = false
                _uiState.value = _uiState.value.copy(isProcessing = false)
            }
        }
    }
    
    fun cancelApply() {
        _uiState.value = _uiState.value.copy(showApplyConfirmation = false)
    }
    
    fun clearLogs() {
        _progressLog.value = emptyList()
    }
    
    private fun addLog(message: String) {
        _progressLog.value = _progressLog.value + message
    }
    
    fun refreshDpMapping() {
        viewModelScope.launch {
            addLog("Fetching DP master list...")
            val result = meroShareRepo.fetchDpMaster()
            if (result.isSuccess) {
                addLog("✅ DP mapping updated")
                // Update UI with new DP codes
                val dps = dpDao.getAllDps()
                _uiState.value = _uiState.value.copy(dpCodes = dps)
            } else {
                addLog("❌ Failed to update DP mapping")
            }
        }
    }
    
    companion object {
        private const val RATE_LIMIT_DELAY = 1500L
    }

## Key Implementation Features

### 1. **BOID Auto-Extraction**
- Regex pattern `\b\d{16}\b` to extract BOIDs from pasted text
- Auto-fills username from the last 8 digits of BOID
- Auto-calculates DP code from positions 4-8

### 2. **MeroShare API Integration**
- Complete authentication flow with JWT tokens
- Bank details retrieval
- Allotment status checking
- IPO application submission

### 3. **Bulk Processing**
- Rate limiting (1.5s delay between requests)
- Progress logging with detailed status updates
- Error handling with specific error messages

### 4. **Security Features**
- Masked PIN and password inputs
- Local storage only (no server uploads)
- Credential validation before saving

### 5. **UI Components**
- Consistent Material 3 design
- Real-time result updates on cards
- Collapsible sections with item counts
- Progress log for bulk operations
