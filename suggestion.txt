## 📊 PERFORMANCE RISK MITIGATION

### 2. PortfolioSummary View Optimization

**Issue**: The "God View" will slow down as transactions grow.

**Solution**: Hybrid Approach - Materialized Views + Incremental Updates

```sql
-- 1. Create denormalized snapshot table
CREATE TABLE PortfolioSnapshot (
    symbol VARCHAR(20) PRIMARY KEY,
    balance INTEGER,
    wacc DECIMAL(15,2),
    invested DECIMAL(15,2),
    lastUpdated DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (symbol) REFERENCES ScripMaster(symbol)
);

-- 2. Incremental update function
CREATE TRIGGER update_portfolio_snapshot
AFTER INSERT ON TransactionHistory
BEGIN
    -- Update snapshot incrementally
    INSERT OR REPLACE INTO PortfolioSnapshot (symbol, balance, wacc, invested, lastUpdated)
    SELECT
        symbol,
        balance,
        wacc,
        invested,
        CURRENT_TIMESTAMP
    FROM PortfolioSummaryView
    WHERE symbol = NEW.symbol;
END;

-- 3. Refresh on demand (for complex recalculations)
CREATE PROCEDURE refresh_portfolio_snapshot()
BEGIN
    -- Full refresh when needed
    REPLACE INTO PortfolioSnapshot
    SELECT * FROM PortfolioSummaryView;
END;
```

**Implementation Strategy**:

```kotlin
class PortfolioManager(
    private val snapshotRepository: PortfolioSnapshotRepository,
    private val transactionRepository: TransactionRepository
) {
    suspend fun getPortfolio(): List<PortfolioItem> {
        // 90% of queries: Use snapshot (fast)
        if (isSnapshotFresh()) {
            return snapshotRepository.findAll()
        }

        // 10% of queries: Full recalc (slow) with caching
        return calculatePortfolioFromTransactions().also {
            snapshotRepository.saveAll(it)
            updateSnapshotTimestamp()
        }
    }

    suspend fun calculatePortfolioFromTransactions(): List<PortfolioItem> {
        // Only for complex recalculations
        return transactionRepository.groupBySymbol()
            .map { (symbol, transactions) ->
                PortfolioItem(
                    symbol = symbol,
                    balance = calculateBalance(transactions),
                    wacc = calculateWACC(transactions),
                    invested = calculateInvested(transactions)
                )
            }
    }
}
```

### 3. Audit Log Optimization

**Issue**: Logging every LTP change creates massive database bloat.

**Solution**: Smart Logging Strategy

```kotlin
class SmartAuditService(
    private val auditRepository: AuditRepository
) {
    // Configurable thresholds
    private val logLevelThresholds = mapOf(
        AuditLevel.DEBUG to 100,      // Log every 100th change
        AuditLevel.INFO to 10,        // Log every 10th change
        AuditLevel.WARNING to 1,      // Log all warnings
        AuditLevel.ERROR to 1         // Log all errors
    )

    suspend fun logLTPChange(symbol: String, oldLTP: Double, newLTP: Double) {
        val changeCount = getChangeCount(symbol)

        // Only log every Nth change for DEBUG level
        if (changeCount % logLevelThresholds[AuditLevel.DEBUG] != 0) {
            // Log to in-memory buffer instead of database
            memoryBuffer.add("LTP: $symbol $oldLTP -> $newLTP")
            return
        }

        // Log significant changes (delta > 5%)
        if (calculateDelta(oldLTP, newLTP) > 5.0) {
            logSignificantChange(symbol, oldLTP, newLTP)
            return
        }

        // Sample logging for regular changes
        auditRepository.save(
            AuditLog(
                entityType = "LTP",
                entityId = symbol,
                action = "UPDATE",
                summary = "LTP changed (delta: ${newLTP - oldLTP})",
                // Store details in compressed format
                details = compressDetails(oldLTP, newLTP)
            )
        )
    }

    // Auto-cleanup strategy
    suspend fun cleanupOldLogs() {
        // Keep last 7 days of detailed logs
        auditRepository.deleteOlderThan(7.days)

        // Keep summarized logs for 30 days
        auditRepository.summarizeOldLogs(olderThan = 30.days) { oldLogs ->
            // Create daily summaries
            createDailySummary(oldLogs)
        }

        // Archive to CSV for historical reference
        if (shouldArchive()) {
            createArchiveCsv()
        }
    }
}
```

**Storage Estimates**:
- **Without optimization**: 200 scrips × 60 min × 24h × 30 days = 8.6 million rows
- **With optimization**: 200 scrips × (1 log/hour + significant changes) × 30 days = ~150,000 rows
- **Storage saved**: 98% reduction

## 🔌 SCRAPING & INTEGRATION RISK MITIGATION

### 4. Scraper Resiliency Architecture

**Issue**: External websites change structure or block scrapers.

**Solution**: Multi-Layer Fallback System

```kotlin
class ResilientScraper(
    private val primarySource: PrimaryScraper,
    private val fallbackSources: List<FallbackScraper>,
    private val cacheManager: CacheManager
) {
    suspend fun scrapeLTPs(): Result<List<LTPData>> {
        // Layer 1: Check cache freshness
        if (cacheManager.isFresh(threshold = 15.minutes)) {
            return Result.success(cacheManager.getCachedLTPs())
        }

        // Layer 2: Primary source (with circuit breaker)
        try {
            val result = primarySource.scrape()
            if (result.isSuccess && result.data.isNotEmpty()) {
                cacheManager.updateCache(result.data)
                return result
            }
        } catch (e: ScraperException) {
            log.warn("Primary scraper failed: ${e.message}")

            // Increment failure counter for circuit breaker
            circuitBreaker.recordFailure()

            // Layer 3: Fallback sources (if circuit breaker is open)
            if (circuitBreaker.isOpen) {
                return tryFallbackSources()
            }
        }

        // Layer 4: Return cached data with stale flag
        if (cacheManager.hasData()) {
            return Result.success(
                data = cacheManager.getCachedLTPs(),
                isStale = true,
                message = "Using cached data (last update: ${cacheManager.lastUpdateTime})"
            )
        }

        // Layer 5: Manual override fallback
        return Result.failure(
            message = "All scrapers failed. Please check your internet connection.",
            canRetry = true,
            manualOverrideAvailable = true
        )
    }

    private suspend fun tryFallbackSources(): Result<List<LTPData>> {
        for (fallback in fallbackSources) {
            try {
                val result = fallback.scrape()
                if (result.isSuccess && result.data.isNotEmpty()) {
                    cacheManager.updateCache(result.data)
                    return result
                }
            } catch (e: Exception) {
                log.warn("Fallback scraper ${fallback.name} failed: ${e.message}")
            }
        }
        return Result.failure("All fallback sources failed")
    }
}
```

### 5. IPO Checking Legal & Technical Compliance

**Issue**: Automated IPO checking may violate CDSC Terms of Service.

**Solution**: Compliance-First Approach

```kotlin
class IpoCheckingService(
    private val rateLimiter: RateLimiter,
    private val cacheManager: IpoCacheManager
) {
    suspend fun checkResults(boids: List<BOID>): List<IpoResult> {
        // 1. Respect CDSC's rate limits
        rateLimiter.waitIfNeeded(rateLimit = 10.seconds)

        // 2. Batch check with user consent
        val userConsent = getUserConsent()
        if (!userConsent) {
            return emptyList()
        }

        // 3. Check cache first (24-hour TTL)
        val cached = cacheManager.getCachedResults(boids)
        val uncached = boids.filter { it !in cached.keys }

        // 4. Check uncached BOIDs with proper delays
        val results = uncached.map { boid ->
            delay(500) // Respectful delay
            checkSingleResult(boid)
        }

        // 5. Cache new results
        cacheManager.cacheResults(results)

        return cached.values + results
    }

    // Provide manual fallback option
    suspend fun showManualCheckInstructions() {
        showDialog(
            title = "IPO Result Check",
            message = """
                For manual checking, visit:
                https://iporesult.cdsc.com.np/result

                Enter your BOID to check manually.

                We'll then help you record the result.
            """.trimIndent(),
            confirmButton = "Open Website",
            cancelButton = "Cancel"
        )
    }
}
```

### 6. Manual vs Automated LTP Conflict Resolution

**Issue**: User manual updates being overwritten by scraper.

**Solution**: Strict Source Enforcement with Visual Indicators

```kotlin
class LtpSourceManager(
    private val ltpRepository: LtpRepository,
    private val scripRepository: ScripRepository
) {
    suspend fun updateLTP(symbol: String, newLTP: Double, source: LtpSource) {
        val scrip = scripRepository.findBySymbol(symbol)
        val current = ltpRepository.findBySymbol(symbol)

        // Asset-specific source enforcement
        when {
            // Tradable assets: Only scraper can update
            scrip.assetType == AssetType.TRADABLE && source != LtpSource.SCRAPER -> {
                throw LtpUpdateException(
                    "Cannot manually update LTP for tradable asset '$symbol'. " +
                    "LTP is automatically updated by the market scraper."
                )
            }

            // Non-tradable assets: Only manual updates allowed
            scrip.assetType != AssetType.TRADABLE && source == LtpSource.SCRAPER -> {
                // Scraper ignores non-tradable assets
                return
            }

            // User override with warning for tradable
            scrip.assetType == AssetType.TRADABLE && source == LtpSource.USER_EDIT -> {
                // Allow but warn
                showWarning(
                    "Manual LTP update for ${scrip.symbol} will be overwritten " +
                    "by the scraper on next market refresh."
                )
            }
        }

        // Save with source tracking
        ltpRepository.save(
            LtpData(
                symbol = symbol,
                ltp = newLTP,
                source = source,
                timestamp = Instant.now()
            )
        )
    }
}
```

### 7. AD/BS Date System Implementation

**Issue**: Off-by-one errors with dual calendar systems.

**Solution**: Unified Internal Storage + Formatting Layer

```kotlin
// 1. Always store in UTC/AD internally
@Entity
data class Transaction(
    @ColumnInfo(name = "transaction_date")
    val transactionDate: LocalDate // Always AD
)

// 2. Separate formatting layer
class DateFormatter(
    private val userPreference: DateSystemPreference
) {
    fun format(date: LocalDate): String {
        return when (userPreference) {
            DateSystem.AD -> formatAD(date)
            DateSystem.BS -> formatBS(date)
        }
    }

    fun parse(input: String): LocalDate {
        return when (userPreference) {
            DateSystem.AD -> parseAD(input)
            DateSystem.BS -> parseBS(input)
        }.also {
            // Validation: Check for off-by-one errors
            validateDate(it)
        }
    }

    private fun validateDate(date: LocalDate) {
        // Additional validation to catch BS conversion errors
        val bsDate = ADtoBS(date)
        val roundTrip = BStoAD(bsDate)

        if (roundTrip != date) {
            log.warning("Date conversion mismatch: $date -> $bsDate -> $roundTrip")
            // Alert user or fallback to AD
        }
    }
}

// 3. Comprehensive testing
class DateFormatterTest {
    @Test
    fun testBSCalendarRoundTrip() {
        // Test all known BS dates
        for (year in 2000..2030) {
            for (month in 1..12) {
                for (day in 1..lastDayOfBSMonth(year, month)) {
                    val adDate = BStoAD(BSDate(year, month, day))
                    val bsDate = ADtoBS(adDate)

                    assertEquals(BSDate(year, month, day), bsDate)
                }
            }
        }
    }
}
```

```kotlin
}
```

### Rollback Strategy

```sql
-- Schema version tracking
CREATE TABLE SchemaVersion (
    version INTEGER PRIMARY KEY,
    appliedAt DATETIME DEFAULT CURRENT_TIMESTAMP,
    migrationName VARCHAR(100),
    rollbackScript TEXT
);

-- Rollback procedure
CREATE PROCEDURE rollback_to_version(targetVersion INTEGER)
BEGIN
    DECLARE currentVersion INTEGER;
    SELECT version INTO currentVersion FROM SchemaVersion ORDER BY version DESC LIMIT 1;

    WHILE currentVersion > targetVersion DO
        EXECUTE (SELECT rollbackScript FROM SchemaVersion WHERE version = currentVersion);
        SET currentVersion = currentVersion - 1;
    END WHILE;
END;
```
