# Revised Finfolio Log Analysis - Bug Report

## Executive Summary
The log reveals **5 major categories of issues** affecting the Finfolio application's data synchronization capabilities. The most critical is the **SSL certificate validation failure** that prevents access to `nepalstock.com`.

---

## 🚨 CRITICAL BUGS

### Bug 1: SSL Certificate Validation Failure
**Severity:** 🔴 Critical
**Frequency:** Repeated across all sync attempts (50+ occurrences)

**Description:**
The application fails to establish HTTPS connections to `https://www.nepalstock.com/` due to missing trust anchors for the certificate chain.

**Error Pattern:**
```
java.security.cert.CertPathValidatorException: Trust anchor for certification path not found.
```

**Impact:**
- Market indices sync fails for this source
- Live price (LTP) sync fails for this source
- Users miss data from what appears to be a primary market data source
- Affects both `MarketSync` and `LtpSync` operations

**Root Cause:**
The Android device's trust store doesn't recognize the certificate authority that signed `nepalstock.com`'s SSL certificate, or the certificate is self-signed/expired.

**Affected URLs:**
- `https://www.nepalstock.com/` (Primary source)
- `https://www.nepalstock.com` (Market Status)

**Timestamp Examples:**
- 2026-06-27 02:20:30.415
- 2026-06-27 02:18:17.176
- 2026-06-27 01:42:41.732

```kotlin
// In LtpSync.kt - Add retry and fallback logic
suspend fun syncWithRetry(maxRetries: Int = 3) {
    repeat(maxRetries) { attempt ->
        try {
            val result = syncLtp()
            if (result.isNotEmpty()) return result
            delay(1000L * (attempt + 1)) // Exponential backoff
        } catch (e: IOException) {
            if (attempt == maxRetries - 1) throw e
            delay(2000L * (attempt + 1))
        }
    }
}

// Implement circuit breaker pattern
class CircuitBreaker {
    private var failureCount = 0
    private val maxFailures = 5
    private val resetTimeout = 60000L
    private var circuitOpenTime = 0L

    suspend fun <T> execute(block: suspend () -> T): T {
        if (isOpen()) {
            if (System.currentTimeMillis() - circuitOpenTime > resetTimeout) {
                // Try to reset
                reset()
            } else {
                throw CircuitOpenException("Circuit is open for this source")
            }
        }
        return try {
            block().also {
                failureCount = 0
                circuitOpenTime = 0
            }
        } catch (e: Exception) {
            failureCount++
            if (failureCount >= maxFailures) {
                circuitOpenTime = System.currentTimeMillis()
            }
            throw e
        }
    }

    private fun isOpen() = failureCount >= maxFailures
    private fun reset() {
        failureCount = 0
        circuitOpenTime = 0
    }
}
```

### Fix 3: Reduce Logging Noise
```kotlin
// In Logging configuration
object SmartLogger {
    private const val MIN_COOLDOWN_LOG_INTERVAL = 60000L // 1 minute
    private val lastCooldownLog = mutableMapOf<String, Long>()

    fun logCooldownIfNeeded(tag: String, source: String) {
        val key = "$tag-$source"
        val now = System.currentTimeMillis()
        val lastLog = lastCooldownLog[key] ?: 0

        if (now - lastLog >= MIN_COOLDOWN_LOG_INTERVAL) {
            Logger.log(Level.DEBUG, tag, "Skipping sync (Cooldown active) for $source")
            lastCooldownLog[key] = now
        }
    }
}

// Use in sync functions
if (isCooldownActive()) {
    SmartLogger.logCooldownIfNeeded("LtpSync", "SCRIP_SYNC")
    return emptyList()
}
```

### Fix 4: Add Health Checks & Source Prioritization
```kotlin
// Add data source health monitoring with user configuration awareness
data class DataSourceHealth(
    val source: String,
    val type: SourceType, // MARKET_INDEX, LTP, IPO, SCRIP
    val isHealthy: Boolean = true,
    val lastSuccessfulSync: Long = 0,
    val consecutiveFailures: Int = 0,
    val isUserConfigured: Boolean = false
)

class SyncHealthMonitor(private val scraperConfig: ScraperConfig) {
    private val healthMap = mutableMapOf<String, DataSourceHealth>()

    fun recordSuccess(source: String, type: SourceType) {
        val key = "$type-$source"
        healthMap[key] = DataSourceHealth(
            source = source,
            type = type,
            isHealthy = true,
            lastSuccessfulSync = System.currentTimeMillis(),
            consecutiveFailures = 0,
            isUserConfigured = scraperConfig.isUserDefined(type, source)
        )
    }

    fun recordFailure(source: String, type: SourceType) {
        val key = "$type-$source"
        val current = healthMap[key] ?: DataSourceHealth(
            source = source,
            type = type,
            isUserConfigured = scraperConfig.isUserDefined(type, source)
        )
        healthMap[key] = current.copy(
            isHealthy = current.consecutiveFailures < 3,
            consecutiveFailures = current.consecutiveFailures + 1
        )
    }

    fun getHealthySources(type: SourceType): List<String> {
        return healthMap.values
            .filter { it.type == type && it.isHealthy }
            .map { it.source }
    }
}
```

### Fix 5: Smart Fallback with User Config Awareness
```kotlin
class SmartSyncManager(
    private val healthMonitor: SyncHealthMonitor,
    private val scraperConfig: ScraperConfig
) {
    suspend fun syncWithFallback(type: SourceType, block: suspend (String) -> Result): SyncResult {
        val userSources = scraperConfig.getUrls(type)
        val healthySources = healthMonitor.getHealthySources(type)

        // Prioritize user-configured sources
        val prioritizedSources = userSources
            .filter { it in healthySources }
            .let {
                if (it.isEmpty()) userSources else it
            }

        for (source in prioritizedSources) {
            try {
                val result = block(source)
                if (result.isSuccess) {
                    healthMonitor.recordSuccess(source, type)
                    return result
                }
            } catch (e: Exception) {
                healthMonitor.recordFailure(source, type)
                // Log but continue to next source
                Logger.log(Level.ERROR, "SyncManager", "Failed on $source: ${e.message}")
            }
        }

        return SyncResult.Error("All sources failed for $type")
    }
}
```
